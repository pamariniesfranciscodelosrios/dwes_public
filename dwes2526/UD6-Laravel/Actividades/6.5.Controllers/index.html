6.5.Controllers

üíª Activity 6.5 Controllers

Stimated time: 1 hour.

  • Migrations with foreign keys (but no Eloquent relationships)
  • Models
  • Blade layout with navigation
  • Standard controller (not invokable)
  • compact() usage
  • /games to list games
  • /games/setup to generate 3 sample genres and 3 sample games using the controller logic

Laravel Guided Project: Basic Video Game App

Requirements:

  • Laravel installed
  • Existing users table (from default auth or previous migration)
  • PHP and Composer working
  • create

Step 1: Create Project, Models and Migrations

Create Models with migrations (-m)

composer create-project laravel/laravel basic-game
cd basic-game
php artisan serve
php artisan make:model Game -m
php artisan make:model Genre -m

üìÑ database/migrations/xxxx_xx_xx_create_genres_table.php

public function up(): void
{
    Schema::create('genres', function (Blueprint $table) {
        $table->id();
        $table->string('name');
        $table->timestamps();
    });
}

üìÑ database/migrations/xxxx_xx_xx_create_games_table.php

public function up(): void
{
    Schema::create('games', function (Blueprint $table) {
        $table->id();
        $table->string('title');
        $table->text('description')->nullable();
        $table->unsignedBigInteger('genre_id');
        $table->unsignedBigInteger('user_id');
        $table->timestamps();

        $table->foreign('genre_id')->references('id')->on('genres')->onDelete('cascade');
        $table->foreign('user_id')->references('id')->on('users')->onDelete('cascade');
    });
}

Step 3: Create Controller

php artisan make:controller GameController

üìÑ app/Http/Controllers/GameController.php

  • As usual, please read and personalize your project, make it yours:
namespace App\Http\Controllers;

use Illuminate\Http\Request;
use App\Models\Game;
use App\Models\Genre;
use App\Models\User;

class GameController extends Controller
{
    public function index()
    {
        $games = Game::all(); // No relationships
        return view('games.index', compact('games'));
    }

    public function setup()
    {
        if (Genre::count() === 0 && Game::count() === 0) {
            $g1 = Genre::create(['name' => 'Action']);
            $g2 = Genre::create(['name' => 'Adventure']);
            $g3 = Genre::create(['name' => 'Puzzle']);

            //create a sample user

            User::create([
                'name' => 'Antonio',
                'email' =>  'antonio@example.com',
                'password' => bcrypt('password123') // Use a secure password
            ]);

           // Assuming you have a User model and at least one user exists
            $user = User::first();

            Game::create([
                'title' => 'Battle Zone',
                'description' => 'An explosive action game.',
                'genre_id' => $g1->id,
                'user_id' => $user->id
            ]);

            Game::create([
                'title' => 'Jungle Quest',
                'description' => 'Explore the mysterious jungle.',
                'genre_id' => $g2->id,
                'user_id' => $user->id
            ]);

            Game::create([
                'title' => 'Brain Teasers',
                'description' => 'Solve complex puzzles.',
                'genre_id' => $g3->id,
                'user_id' => $user->id
            ]);
        }

        return redirect()->route('games.index')->with('message', 'Sample data created!');
    }
}

  • Game: all

Step 4: Define Routes

üìÑ routes/web.php

Important: use ...

use Illuminate\Support\Facades\Route;
//import the GameController, which contains the logic for handling game-related requests
use App\Http\Controllers\GameController;

// Route to display all games
Route::get('/games', [GameController::class, 'index'])->name('games.index');
// Route to setup initial data
Route::get('/games/setup', [GameController::class, 'setup'])->name('games.setup');
